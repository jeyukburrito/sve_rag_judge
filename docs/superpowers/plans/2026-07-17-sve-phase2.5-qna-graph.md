```python
import argparse
import csv
import glob
import json
import re
import sys
from pathlib import Path

import chromadb

from src.build_index import DEFAULT_DB, index_documents

ROOT = Path(__file__).resolve().parent.parent
CARDDB_GLOB = str(ROOT / "data" / "carddb" / "*.csv")
QNA_FILE = ROOT / "data" / "qna" / "unique_qna.jsonl"
GRAPH_FILE = ROOT / "build" / "graph.json"
CARD_REF_RE = re.compile(r"『([^』]+)』")


def card_doc(row: dict) -> tuple[str, dict]:
    text = (
        f"{row['card_code']} {row['name']} / クラス:{row['class']} / "
        f"{row['card_type']}({row['sub_type']}) / "
        f"コスト{row['cost']} 攻{row['attack']} 体{row['hp']} / "
        f"効果: {row['effect']}"
    )
    meta = {k: row[k] for k in ("card_code", "name", "class", "card_type")}
    return text, meta


def qna_doc(rec: dict) -> tuple[str, dict]:
    text = f"Q: {rec['question']}\nA: {rec['answer']}"
    meta = {
        "qa_id": rec["qa_id"],
        "category": rec["category"] or "",
        "cards": ",".join(c["card_code"] for c in rec["cards"]),
        "date": rec["date"],
    }
    return text, meta


def read_card_rows() -> list[dict]:
    rows: list[dict] = []
    for path in sorted(glob.glob(CARDDB_GLOB)):
        with open(path, encoding="utf-8-sig", newline="") as f:
            rows.extend(csv.DictReader(f))
    return rows


def read_qna_recs() -> list[dict]:
    recs: list[dict] = []
    with open(QNA_FILE, encoding="utf-8") as f:
        for line in f:
            if line.strip():
                recs.append(json.loads(line))
    return recs


def load_cards() -> list[tuple[str, dict]]:
    return [card_doc(r) for r in read_card_rows()]


def load_qna() -> list[tuple[str, dict]]:
    return [qna_doc(r) for r in read_qna_recs()]


def build_graph(rows: list[dict], recs: list[dict]) -> dict:
    name_to_codes: dict[str, list[str]] = {}
    code_to_name: dict[str, str] = {}
    for r in rows:
        name_to_codes.setdefault(r["name"], []).append(r["card_code"])
        code_to_name[r["card_code"]] = r["name"]

    code_to_qa: dict[str, list[str]] = {}
    qa_dates: dict[str, str] = {}
    for rec in recs:
        qa_dates[rec["qa_id"]] = rec.get("date", "")
        for c in rec["cards"]:
            code_to_qa.setdefault(c["card_code"], []).append(rec["qa_id"])

    code_to_refs: dict[str, list[str]] = {}
    for r in rows:
        refs = [
            n for n in dict.fromkeys(CARD_REF_RE.findall(r.get("effect", "") or ""))
            if n in name_to_codes and n != r["name"]
        ]
        if refs:
            code_to_refs[r["card_code"]] = refs

    return {
        "code_to_qa": code_to_qa,
        "name_to_codes": name_to_codes,
        "code_to_name": code_to_name,
        "code_to_refs": code_to_refs,
        "qa_dates": qa_dates,
    }


def main() -> None:
    sys.stdout.reconfigure(encoding="utf-8")
    p = argparse.ArgumentParser(description="카드 DB와 공식 Q&A를 ChromaDB에 인덱싱 + 관계 맵 생성")
    p.add_argument("--db", default=DEFAULT_DB)
    p.add_argument("--graph-only", action="store_true", help="임베딩 없이 관계 맵만 생성")
    args = p.parse_args()

    rows = read_card_rows()
    recs = read_qna_recs()

    graph = build_graph(rows, recs)
    GRAPH_FILE.parent.mkdir(parents=True, exist_ok=True)
    GRAPH_FILE.write_text(json.dumps(graph, ensure_ascii=False), encoding="utf-8")
    print(f"관계 맵: {GRAPH_FILE} (code_to_qa {len(graph['code_to_qa'])}종)")
    if args.graph_only:
        return

    client = chromadb.PersistentClient(path=args.db)
    cards = [card_doc(r) for r in rows]
    print(f"카드 {len(cards)}건 로드")
    index_documents(client, "cards", cards)
    qna = [qna_doc(r) for r in recs]
    print(f"Q&A {len(qna)}건 로드")
    index_documents(client, "qna", qna)
    print(f"완료: cards={len(cards)}, qna={len(qna)} → {args.db}")


if __name__ == "__main__":
    main()
```

```python
import json
import re
import sys
from pathlib import Path

import ollama
from langchain_chroma import Chroma
from langchain_core.messages import HumanMessage, SystemMessage
from langchain_ollama import ChatOllama, OllamaEmbeddings

EMBED_MODEL = "bge-m3"
DEFAULT_LLM = "qwen3:8b"
# cwd가 아니라 리포 루트 기준으로 고정 — src/ 안에서 실행해도 같은 인덱스를 쓴다

GRAPH_FILE = str(ROOT / "build" / "graph.json")
MAX_DISTANCE = 0.75  # cosine distance. 소스별 동일 게이트

LINKED_QA_PER_CARD = 6   # 카드당 연결 재정 상한 (날짜 최신순)
LINKED_QA_TOTAL = 12     # 연결 재정 전체 상한
TOKEN_REF_DOCS = 2       # 토큰/참조 카드 문서 상한
JOIN_CARD_HITS = 2       # 임베딩 카드 히트 중 조인 대상 상위 수
# \b is Unicode-aware in Python 3, so a code touching a Korean/Japanese particle
# (e.g. "BP01-001과") has no word boundary and would not match. Use ASCII-only
# boundary lookarounds so CJK-adjacent codes still match, without over-matching.
CARD_CODE_RE = re.compile(r"(?<![0-9A-Za-z])[A-Za-z]{2,4}\d{2}[A-Za-z]?-\d{1,3}(?![0-9A-Za-z])")

_EMPTY_HINTS = {

def ensure_ollama(models: list[str]) -> None:
    try:
        installed = [m.model for m in ollama.list().models]
    except Exception:
        sys.exit("Ollama에 연결할 수 없습니다. 설치/실행 후 재시도: https://ollama.com")
    for model in models:
        if ":" in model:
            ok = model in installed
        else:
            ok = any(name.split(":")[0] == model for name in installed)
        if not ok:
            sys.exit(f"모델이 없습니다. 먼저 실행하세요: ollama pull {model}")


def extract_card_codes(text: str) -> list[str]:
    seen: list[str] = []
    for code in CARD_CODE_RE.findall(text):
        if code not in seen:
            seen.append(code)
    return seen


def load_graph(path: str = GRAPH_FILE) -> dict:
    p = Path(path)
    if not p.exists():
        sys.exit(
            f"관계 맵이 없습니다: {p}\n"
            "먼저 실행하세요: python -m src.build_phase2_index --graph-only"
        )
    return json.loads(p.read_text(encoding="utf-8"))


def open_stores(db: str) -> dict:
    """세 컬렉션 Chroma 스토어 + 관계 맵("graph" 키)을 반환한다."""
    emb = OllamaEmbeddings(model=EMBED_MODEL)
    stores: dict = {}
    for name in ("rules", "cards", "qna"):
        store = Chroma(collection_name=name, persist_directory=db, embedding_function=emb)
        if store._collection.count() == 0:
            sys.exit(
                f"'{name}' 인덱스가 비어 있습니다: {db}\n"
                f"먼저 실행하세요: {_EMPTY_HINTS[name]}"
            )
        stores[name] = store
    stores["graph"] = load_graph()
    return stores


def linked_qa_ids(
    codes: list[str], graph: dict,
    per_card: int = LINKED_QA_PER_CARD, total: int = LINKED_QA_TOTAL,
) -> list[str]:
    """코드들을 동명 그룹으로 확장해 연결 재정 id를 최신순·상한 적용으로 모은다."""
    out: list[str] = []
    seen: set[str] = set()
    for code in codes:
        name = graph["code_to_name"].get(code)
        group = graph["name_to_codes"].get(name, [code]) if name else [code]
        ids: list[str] = []
        for c in group:
            ids.extend(graph["code_to_qa"].get(c, []))
        ids = sorted(set(ids), key=lambda q: graph["qa_dates"].get(q, ""), reverse=True)[:per_card]
        for q in ids:
            if q not in seen:
                seen.add(q)
                out.append(q)
                if len(out) >= total:
                    return out
    return out


def token_ref_names(codes: list[str], graph: dict, limit: int = TOKEN_REF_DOCS) -> list[str]:
    out: list[str] = []
    for code in codes:
        for n in graph["code_to_refs"].get(code, []):
            if n not in out:
                out.append(n)
                if len(out) >= limit:
                    return out
    return out


def _label(source: str, meta: dict) -> str:
    if source == "rules":
        return f"룰 {meta['rule_id']}"
    if source == "cards":
        return f"카드 {meta['card_code']} {meta['name']}"
    return f"Q&A {meta['qa_id']}"


def retrieve_all(stores: dict, question: str) -> list[tuple[str, str, float]]:
    graph = stores["graph"]
    hits: list[tuple[str, str, float]] = []
    seen_labels: set[str] = set()

    def add(label: str, doc: str, dist: float) -> None:
        if label not in seen_labels:
            seen_labels.add(label)
            hits.append((label, doc, dist))

    # 1) 질문 속 카드 코드 정확 조회
    exact_codes: list[str] = []
    for code in extract_card_codes(question):
        got = stores["cards"].get(where={"card_code": code})
        if not got["documents"]:
            # 스펙: 미발견 코드는 무시하되 컨텍스트에 표시
            add(f"카드 {code}", "카드 DB에서 이 코드를 찾지 못했습니다.", 0.0)
            continue
        exact_codes.append(code)
        for doc, meta in zip(got["documents"], got["metadatas"]):
            add(_label("cards", meta), doc, 0.0)

    # 2) 소스별 쿼터 임베딩 검색 (수집만 — 배치는 마지막에)
    embed_hits: list[tuple[str, str, float]] = []
    embed_card_codes: list[str] = []
    for source, k in QUOTAS.items():
        for doc, dist in stores[source].similarity_search_with_score(question, k=k):
            if dist > MAX_DISTANCE:
                continue
            if source == "cards" and len(embed_card_codes) < JOIN_CARD_HITS:
                code = doc.metadata["card_code"]
                if code not in embed_card_codes and code not in exact_codes:
                    embed_card_codes.append(code)
            embed_hits.append((_label(source, doc.metadata), doc.page_content, dist))

    # 3) 연결 재정 조인 — 컨텍스트 선두(정확조회 카드 바로 뒤)에 배치
    qa_ids = linked_qa_ids(exact_codes + embed_card_codes, graph)
    if qa_ids:
        got = stores["qna"].get(where={"qa_id": {"$in": qa_ids}})
        order = {q: i for i, q in enumerate(qa_ids)}
        pairs = sorted(
            zip(got["documents"], got["metadatas"]),
            key=lambda p: order.get(p[1]["qa_id"], len(order)),
        )
        for doc, meta in pairs:
            add(f"Q&A {meta['qa_id']}", doc, 0.0)

    # 4) 토큰/참조 카드 문서 (동명 재판은 1장만)
    ref_names = token_ref_names(exact_codes + embed_card_codes[:1], graph)
    if ref_names:
        got = stores["cards"].get(where={"name": {"$in": ref_names}})
        used: set[str] = set()
        for doc, meta in zip(got["documents"], got["metadatas"]):
            if meta["name"] in used:
                continue
            used.add(meta["name"])
            add(_label("cards", meta), doc, 0.0)

    # 5) 임베딩 히트는 마지막에 (연결 재정과 중복되면 자동 제거)
    for label, doc, dist in embed_hits:
        add(label, doc, dist)
    return hits


def answer(llm: ChatOllama, question: str, hits: list[tuple[str, str, float]]) -> str:
    context = "\n\n".join(f"[{label}] {doc}" for label, doc, _ in hits)
    res = llm.invoke([
        SystemMessage(content=SYSTEM_PROMPT),
        HumanMessage(content=f"근거 자료:\n{context}\n\n질문: {question}"),
    ])
    return res.content


def main() -> None:
    sys.stdout.reconfigure(encoding="utf-8")
    p = argparse.ArgumentParser(description="SVE 판정 저지 CLI")
    p.add_argument("--db", default=DEFAULT_DB)
    p.add_argument("--llm", default=DEFAULT_LLM)
    args = p.parse_args()

    ensure_ollama([EMBED_MODEL, args.llm])
    stores = open_stores(args.db)
    llm = ChatOllama(model=args.llm)

    print("SVE 판정 저지 — 질문을 입력하세요 (종료: 빈 줄)")
    while True:
        try:
            question = input("\n질문> ").strip()
        except (EOFError, KeyboardInterrupt):
            break
        if not question:
            break
        hits = retrieve_all(stores, question)
        if not hits:
            print("관련 근거를 찾지 못했습니다.")
            continue
        print("\n" + answer(llm, question, hits))
        print("\n--- 참조 근거 ---")
        for label, doc, dist in hits:
            print(f"[{label}] (거리 {dist:.2f}) {doc[:120]}")


if __name__ == "__main__":
    main()
```
